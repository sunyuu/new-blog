function plugin() {

function tokenizeMention(eat, value, silent) {
    if (value.charAt(0) !== '@') {
        return
    }

    var match = /@\[(\w+)\]\((\S+)(?:\"(\S+)\")?(?:\"(\S+)\")?\)/.exec(value);

    if (match) {
        console.log(match)

            return true;

            type: 'link',
            url: match[2],
            title: match[3] ? match[3] : null,
            data: {
                hProperties: {
                    className: 'cardlink'
                }
            },
            children: [
                {

}

function locateMention(value, fromIndex) {
